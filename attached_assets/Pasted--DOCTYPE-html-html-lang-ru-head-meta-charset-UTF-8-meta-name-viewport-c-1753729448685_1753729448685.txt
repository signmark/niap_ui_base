    <script>
        document.getElementById('authForm').addEventListener('submit', async (e) => {
            e.preventDefault();

            const appId = document.getElementById('appId').value;
            const appSecret = document.getElementById('appSecret').value;
            const webhookUrl = document.getElementById('webhookUrl').value;
            const instagramId = document.getElementById('instagramId').value;

            // Правильно формируем URLs для облачной среды с прокси
            let baseUrl = window.location.origin;
            if (baseUrl.endsWith('/')) {
                baseUrl = baseUrl.slice(0, -1); // Убираем слеш в конце
            }

            const apiUrl = baseUrl + '/api/auth/start';
            const redirectUri = baseUrl + '/api/auth/callback';

            console.log('Base URL:', baseUrl);
            console.log('API URL:', apiUrl);
            console.log('Redirect URI:', redirectUri);

            try {
                const response = await fetch(apiUrl, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ appId, appSecret, redirectUri, webhookUrl, instagramId })
                });

                console.log('Response status:', response.status);

                if (!response.ok) {
                    const errorText = await response.text();
                    console.log('Error response:', errorText);
                    showResult(`HTTP ${response.status}: ${errorText}`, 'error');
                    return;
                }

                const data = await response.json();

                if (data.authUrl) {
                    console.log('Redirecting to Facebook:', data.authUrl);
                    window.location.href = data.authUrl;
                } else {
                    showResult(data.error || 'Неизвестная ошибка', 'error');
                }
            } catch (error) {
                console.error('Error:', error);
                showResult('Ошибка: ' + error.message, 'error');
            }
        });

        function showResult(message, type) {
            const resultDiv = document.getElementById('result');
            resultDiv.innerHTML = `<div class="${type}">${message}</div>`;
        }
    </script>
